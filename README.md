# Quick Installation

    git clone https://github.com/imadityaim/Sophia-AI-Desktop-Assistnt

# Create Environment

    conda create -p venv python

# Environment Activate

    conda activate venv/

# Install All Package

    pip install -r requirements.txt  
    
# Run The Application

    python run.py 

# Note:
You can Directly run this WebApp on VS Code.
